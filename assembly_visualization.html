def grouped_bar_plot(df, c, a, nr, color, description):
    p = sns.barplot(data = df, x = 'sample_name', y = c, ax = a[nr], color= color ) 
    p.tick_params(axis='x', labelrotation=90, size = 40)
    p.set_ylabel(description, size = 40)
    p.set_xlabel('Samples Names', size = 40)
    #p.get_legend().remove()
    return p
